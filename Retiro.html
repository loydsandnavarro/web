        function calcularResultado() {

            let puntuacion = 0;

            let respuestas = document.querySelectorAll("input[type='radio']:checked");

            respuestas.forEach(radio => {

                puntuacion += parseInt(radio.value);

            });

            

            let resultadoTexto = "";

            if (puntuacion >= 13) {

                resultadoTexto = "¡Felicidades! 🎉 Estás bien encaminado hacia un retiro seguro y cómodo.";

            } else if (puntuacion >= 8) {

                resultadoTexto = "Vas por buen camino, pero aún hay ajustes por hacer.";

            } else if (puntuacion >= 4) {

                resultadoTexto = "Es momento de actuar. Tu preparación para el retiro es baja.";

            } else {

                resultadoTexto = "🚨 ¡Urgente! Necesitas un plan YA.";

            }

            

            document.getElementById("resultado").innerText = `Puntuación: ${puntuacion}\n${resultadoTexto}`;

            document.getElementById("asesoria").style.display = "block";

        }

    </script>
